<script>
    // 1. Handle Login Transition
    function showApp() {
        const email = document.getElementById('email').value;
        if(!email.includes('@')) {
            alert("Please enter a valid email");
            return;
        }
        document.getElementById('authSection').classList.add('hidden');
        document.getElementById('mainApp').classList.remove('hidden');
    }

    // 2. Handle Image Preview
    function previewFile() {
        const preview = document.getElementById('userImg');
        const file = document.getElementById('imageInput').files[0];
        const reader = new FileReader();

        reader.onloadend = function () {
            preview.src = reader.result;
            preview.classList.remove('hidden');
            document.getElementById('preText').classList.add('hidden');
        }

        if (file) { reader.readAsDataURL(file); }
    }

    // 3. The "AI" Logic & Personalized Routine
    function runAI() {
        const porosity = document.getElementById('porositySelect').value;
        const imgCheck = document.getElementById('imageInput').files[0];

        if (!imgCheck || !porosity) {
            alert("Please upload a photo and select porosity!");
            return;
        }

        // Simulate AI Thinking
        const btn = document.querySelector('button[onclick="runAI()"]');
        btn.innerText = "Analyzing Texture Patterns...";
        
        setTimeout(() => {
            btn.classList.add('hidden');
            const results = document.getElementById('results');
            const routine = document.getElementById('resRoutine');
            const brands = document.getElementById('brandList');
            
            results.classList.remove('hidden');

            if (porosity === "high") {
                routine.innerHTML = "<strong>Detected: Type 4 High Porosity.</strong><br>Your cuticles are open. Use the <strong>L.C.O. Method</strong>. Seal with heavy butters to trap moisture.";
                brands.innerHTML = "<span class='brand-tag'>TGIN</span> <span class='brand-tag'>Camille Rose</span> <span class='brand-tag'>Adwoa Beauty</span>";
            } else {
                routine.innerHTML = "<strong>Detected: Type 4 Low Porosity.</strong><br>Your cuticles are tightly closed. Use the <strong>L.O.C. Method</strong>. Apply products with steam or heat to help penetration.";
                brands.innerHTML = "<span class='brand-tag'>Mielle Organics</span> <span class='brand-tag'>Pattern Beauty</span> <span class='brand-tag'>Melanin Haircare</span>";
            }
        }, 2000);
    }
</script>
